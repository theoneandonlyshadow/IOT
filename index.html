▶ Program 6: Arduino program to implement Interactive Traffic System using Push Button. 

              const int redLED = 13;
              const int yellowLED = 12;
              const int greenLED = 11;
              const int buttonPin = 2;
              // Variable to store button state
              int buttonState = 0;
              // Variable to track traffic light state
              int trafficState = 0;
              void setup() {
                // Set LED pins as OUTPUT
                pinMode(redLED, OUTPUT);
                pinMode(yellowLED, OUTPUT);
                pinMode(greenLED, OUTPUT);
                // Set button pin as INPUT
                pinMode(buttonPin, INPUT);
                // Initialize serial communication for debugging
                Serial.begin(9600);
                // Initialize with red light ON
                digitalWrite(redLED, HIGH);
                digitalWrite(yellowLED, LOW);
                digitalWrite(greenLED, LOW);
              }
              void loop() {
               // Read the state of the push button
                buttonState = digitalRead(buttonPin);
                // If button is pressed, move to the next traffic state
              if (buttonState == HIGH) {
                delay(300); // Debouncing delay
                trafficState++;
                if (trafficState > 2) {
                  trafficState = 0; // Reset traffic state after 3 stages
                }
                // Control LEDs based on traffic light state
                switch (trafficState) {
                  case 0: // Red light ON
                  digitalWrite(redLED, HIGH);
                  digitalWrite(yellowLED, LOW);
                  digitalWrite(greenLED, LOW);
                  Serial.println("Red Light");
                  break;

                  case 1: // Green light ON
                    digitalWrite(redLED, LOW);
                    digitalWrite(yellowLED, LOW);
                    digitalWrite(greenLED, HIGH);
                    Serial.println("Green Light");
                    break;

                  case 2: // Yellow light ON
                    digitalWrite(redLED, LOW);
                    digitalWrite(yellowLED, HIGH);
                    digitalWrite(greenLED, LOW);
                    Serial.println("Yellow Light");
                    break;
                  }
                  delay(1000); // Wait 1 second between state transitions
                }
              }

                  myservo.attach(9); // Attach the servo to digital pin 9

                }
