▶ Program 4: Interface Push button and LEDs on Arduino Uno 
const int ledPins[] = {2, 3, 4};
const int buttonPin = 5;
int buttonState = 0;
int ledState = 0;

void setup() {
  Serial.begin(9600);
  for (int i = 0; i < 3; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
  pinMode(buttonPin, INPUT);
}

void loop() {
  buttonState = digitalRead(buttonPin);
  if (buttonState == HIGH) {
    ledState = (ledState + 1) % 3;
    switch (ledState) {
      case 0:
        digitalWrite(ledPins[0], HIGH);
        digitalWrite(ledPins[1], LOW);
        digitalWrite(ledPins[2], LOW);
        break;
      case 1:
        digitalWrite(ledPins[0], LOW);
        digitalWrite(ledPins[1], HIGH);
        digitalWrite(ledPins[2], LOW);
        break;
      case 2:
        digitalWrite(ledPins[0], LOW);
        digitalWrite(ledPins[1], LOW);
        digitalWrite(ledPins[2], HIGH);
        break;
    }
    Serial.println("Traffic light state changed");
  }
  delay(100);
}

SoftwareSerial bluetooth(2, 3);

  Serial.begin(9600);

}

    char c = bluetooth.read();

      digitalWrite(13, LOW);
      Serial.println("LED turned off");
    }
  }
  delay(100);
}

  Serial.begin(9600);
  SPI.begin();

      Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ");

    }
    Serial.println();
    Serial.println("Access granted");
  }

}

▶ Program 11: Interface Ultra Sonic HC-SR04 on Raspberry Pi 
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
TRIG = 23
ECHO = 24

GPIO.setup(TRIG, GPIO.OUT)
GPIO.setup(ECHO, GPIO.IN)

while True:
    GPIO.output(TRIG, True)
    time.sleep(0.00001)
    GPIO.output(TRIG, False)
    start = time.time()
    while GPIO.input(ECHO) == 0:
        start = time.time()
    while GPIO.input(ECHO) == 1:
        end = time.time()
    distance = (end - start) * 17150
    print('Distance: {0:0.1f}cm'.format(distance))
    time.sleep(1)

▶ Program 13: Demonstrate the use of ESP8266 for the temperature sensor in a Thingspeak cloud 
#include 
#include 

const char* ssid = "your_ssid";
const char* password = "your_password";
const char* thingspeak_api_key = "your_api_key";

WiFiClient client;
HTTPClient http;

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  Serial.println("Connected to WiFi");
}

void loop() {
  int temperature = analogRead(A0);
  temperature = map(temperature, 0, 1023, 0, 100);
  http.begin(client, "http://api.thingspeak.com/update?api_key=" + String(thingspeak_api_key) + "&field1=" + String(temperature));
  int httpCode = http.GET();
  if (httpCode == HTTP_CODE_OK) {
    String response = http.getString();
    Serial.println(response);
  } else {
    Serial.println("Error sending data to Thingspeak");
  }
  http.end();
  delay(10000);
}
